def to_feed(feed)
  xss = ::RSS::XMLStyleSheet.new
  guess_type_if_need(xss)
  set = setup_values(xss)
  if set
    feed.xml_stylesheets << xss
  end
end

def guess_type_if_need(xss)
  if @type.nil?
    xss.href = @href
    @type = xss.type
  end
end
