def inspector=(inspector)
  case inspector
  when :pp
    raise 'lazy_inspection option must be enabled when using pp' unless lazy_inspection
    require 'pp'
    if colorize_message
      require 'irb/color_printer'
    end
  when :p
  else
    raise ArgumentError, "unknown inspector: #{inspector}"
  end
  super
end
