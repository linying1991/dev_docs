def omit(message=nil, &block)
  message ||= "omitted."
  if block_given?
    omission = Omission.new(name, filter_backtrace(caller), message,
                            :method_name => @method_name)
    add_omission(omission)
  else
    raise OmittedError.new(message)
  end
end
